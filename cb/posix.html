Useful POSIX Recipes

1. Spawn Process (a.k.a. fork-exec)

http://en.wikipedia.org/wiki/Fork_(system_call)

#include <unistd.h>

void
spawn(char *const argv[])
{
    pid_t pid;
    int status;

    pid = fork();
    if (pid > 0) {
        /* fork() succeeded. */
        waitpid(pid, &status, 0);
    } else if (pid == 0) {
        /* Child process. */
        execvp(argv[0], argv);
    }
}

3. List Directory

#include <stdio.h>
#include <dirent.h>

int
ls(const char *path)
{
    DIR *dp;
    struct dirent *ep;

    if(!(dp = opendir(path)))
        return -1;
    while ((ep = readdir(dp)))
        puts(ep->d_name);
    closedir(dp);
    return 0;
}
